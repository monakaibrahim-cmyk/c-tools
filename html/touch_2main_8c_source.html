126int recursive_mkdir(const char* path)
127{
128 char temp[MAX_PATH];
129 char* p = NULL;
130 size_t length;
131
132 /* Copy the input path to a temporary buffer to avoid modifying the original */
133 SAFE_STRNCPY(temp, path, MAX_PATH);
134 length = strlen(temp);
135
136 /* Normalize all path separators to backslashes for Windows */
137 for (p = temp; *p; p++)
138 {
139 if (*p == '/' || *p == '\\')
140 {
141 *p = '\\';
142 }
143 }
144
145 /* Walk through the path and create each directory component */
146 for (p = temp + 1; *p; p++)
147 {
148 if (*p == '\\')
149 {
150 /* Temporarily terminate the string at this position */
151 *p = '\0';
152
153 /* Attempt to create the directory */
154 if (_mkdir(temp) != 0 && errno != EEXIST)
155 {
156 /* Restore the separator and return error */
157 *p = '\\';
158 return -1;
159 }
160
161 /* Restore the separator and continue */
162 *p = '\\';
163 }
164 }
165
166 /* Create the final directory in the path */
167 if (_mkdir(temp) != 0 && errno != EEXIST)
168 {
169 return -1;
170 }
171
172 return 0;
173}

190int _absolute(const char* path)
191{
192 /* Handle NULL or too-short strings as absolute for safety */

205 if (path[0] == '\\' && path[1] == '\\')

228void _timestamps(HANDLE hFile, Options* options)
229{
230 FILETIME access, write;
231 SYSTEMTIME system_time;
232
233 /* Determine the source of timestamps based on options */
234 if (options->_ref_file)
235 {
236 /* Open the reference file to copy its timestamps */
237 HANDLE hRef = CreateFile(
238 options->_ref_file,
239 GENERIC_READ,
240 FILE_SHARE_READ,
241 NULL, OPEN_EXISTING,
242 FILE_ATTRIBUTE_NORMAL,
243 NULL
244 );
245
246 if (hRef != INVALID_HANDLE_VALUE)
247 {
248 /* Read timestamps from the reference file */
249 GetFileTime(hRef, NULL, &access, &write);
250 CloseHandle(hRef);
251 }
252 }
253 else if (options->_timestamp)
254 {
255 /* Parse the explicit timestamp string (format: YYYYMMDDhhmm) */
256 SAFE_SSCANF(
257 options->_timestamp,
258 "%04hd%02hd%02hd%02hd%02hd",
259 &system_time.wYear,
260 &system_time.wMonth,
261 &system_time.wDay,
262 &system_time.wHour,
263 &system_time.wMinute
264 );
265 /* Convert system time to file time format */
266 SystemTimeToFileTime(&system_time, &access);
267 write = access;
268 }
269 else
270 {
271 /* Use current system time for both timestamps */
272 GetSystemTimeAsFileTime(&access);
273 write = access;
274 }
275
276 /* Apply timestamps based on options (atime_only, mtime_only, or both) */
277 FILETIME *pAccess = options->_mtime_only ? NULL : &access;
278 FILETIME *pWrite = options->_atime_only ? NULL : &write;
279
280 /* Set the file timestamps */
281 SetFileTime(hFile, NULL, pAccess, pWrite);
282}
