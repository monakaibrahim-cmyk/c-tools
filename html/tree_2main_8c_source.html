100void list_dir(const char* base, const char* prefix, int* file_count, int* dir_count)
101{
102 /* Construct search pattern: base_path\* to enumerate all entries */
103 char search_path[MAX_PATH];
104 int sp_len = snprintf(search_path, MAX_PATH, "%s\\*", base);
105
106 if (sp_len < 0 || sp_len >= MAX_PATH)
107 {
108 return;
109 }
110
111 /* Initialize Windows FindFirstFile to begin enumeration */
112 WIN32_FIND_DATA ffd;
113 HANDLE hFind = FindFirstFile(search_path, &ffd);
114
115 if (hFind == INVALID_HANDLE_VALUE)
116 {
117 return;
118 }
119
120 /* Build a linked list of directory entries first (batch collection)
121 * This separates enumeration from processing, allowing sorted output */
122 Node *head = NULL, *tail = NULL;
123
124 do
125 {
126 /* Skip special directory references: "." (current) and ".." (parent) */
127 if (strcmp(ffd.cFileName, ".") == 0 || strcmp(ffd.cFileName, "..") == 0)
128 {
129 continue;
130 }
131
132 /* Allocate new node to store this directory entry */
133 Node* new_node = (Node*)malloc(sizeof(Node));
134
135 if (!new_node)
136 {
137 fprintf(stderr, "Error: Memory exhausted while enumerating " YELLOW "'%s' " RESET ". Output may be incomplete.\n", base);
138 break;
139 }
140
141 new_node->data = ffd;
142 new_node->next = NULL;
143
144 /* Append to linked list (tail insertion for O(1) append) */
145 if (!head)
146 {
147 head = tail = new_node;
148 }
149 else
150 {
151 tail->next = new_node;
152 tail = new_node;
153 }
154 }
155 while (FindNextFile(hFind, &ffd));
156
157 /* Close the find handle - no longer needed after enumeration */
158 FindClose(hFind);
159
160 /* Process each entry in the linked list and display tree structure */
161 Node* current = head;
162
163 while (current)
164 {
165 /* Determine if this is the last entry for proper tree branch formatting:
166 * ├── = not last, └── = last */
167 int is_last = (current->next == NULL);
168
169 /* Print the tree branch prefix + filename in green color */
170 printf("%s%s " GREEN "%s" RESET "\n", prefix, is_last ? "└──" : "├──", current->data.cFileName);
171
172 /* Check if entry is a directory (vs file) using Windows attribute flags */
173 if (current->data.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY)
174 {
175 (*dir_count)++;
176
177 /* Construct full path for recursive descent into subdirectory */
178 char next_path[MAX_PATH];
179 int path_len = snprintf(next_path, MAX_PATH, "%s\\%s", base, current->data.cFileName);
180
181 if (path_len < 0 || path_len >= MAX_PATH)
182 {
183 fprintf(stderr, "Error: Path too long: " YELLOW "'%s'" RESET ". Skipping.\n", current->data.cFileName);
184 }
185 else
186 {
187 /* Calculate prefix for next recursion level:
188 * - If last entry: use spaces " " (no vertical branch)
189 * - If not last: use "│" (vertical branch continues) */
190 int probe = snprintf(NULL, 0, "%s%s ", prefix, is_last ? " " : "│");
191
192 if (probe < 0)
193 {
194 fprintf(stderr, "Error: Encoding failed for prefix at " YELLOW "'%s'" RESET ". Skipping.\n", current->data.cFileName);
195 }
196 else
197 {
198 size_t needed = (size_t)probe + 1;
199 char* next_prefix = malloc(needed);
200
201 if (next_prefix)
202 {
203 /* Format the prefix and recurse into subdirectory */
204 snprintf(next_prefix, needed, "%s%s ", prefix, is_last ? " " : "│");
205 list_dir(next_path, next_prefix, file_count, dir_count);
206 free(next_prefix);
207 }
208 else
209 {
210 fprintf(stderr, "Error: Memory exhausted at " YELLOW "'%s'" RESET ".\n", current->data.cFileName);
211 }
212 }
213 }
214 }
215 else
216 {
217 /* Entry is a file - increment file counter */
218 (*file_count)++;
219 }
220
221 current = current->next;
222 }
223
224 /* Clean up linked list memory to prevent leaks */
225 free_list(head);
226}
